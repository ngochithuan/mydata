Table: Users
-> thuộc tính đê phân biệt Admin, Staff (Receptionist, House keeper, Parking Staff), Customer
+ userId
+ fullname (required)
+ email?
+ phone (required)
+ isStaff
+ roleName (normalStaff, cleaningStaff, guard, admin)


Table: Rooms
+ roomId
+ roomTitle
+ roomTypeId
+ roomDescription
+ roomImgs
+ roomStatus (available, checkedin, pending -> để dọn cbi phòng)


Table: RoomType
+ roomTypeId
+ Description
+ Features
+ Amenities
+ Img
+ Price


Table: Bookings (theo lần nghỉ/đặt)
+ bookingId (PK)
+ userId? (nullable, nếu user đã login)
+ fullname? (nullable)
+ email? (nullable)
+ phone? (nullable)
+ checkInDate
+ checkOutDate
+ totalPrice
+ paymentStatus


Table: BookingDetails (theo từng phòng)
+ bookingDetailId (PK)
+ bookingId (FK -> Bookings.bookingId)
+ roomId (FK -> Rooms.roomId)
+ pricePerNight (lưu snapshot giá tại thời điểm book)
+ roomPriceTotal (phòng này bao nhiêu đêm * giá)


Table: Reviews (theo lần nghỉ/đặt)
+ reviewId
+ roomId
+ userId
+ bookingId
+ reviewContent
+ rating


Table: RoomService (theo từng phòng)
+ roomServiceId
+ roomId
+ datetime
+ isCleaningDone (= false default)


Table: ParkingService (theo lần nghỉ/đặt)
+ parkingServiceId
+ bookingId
+ parkingPlateNo = VARCHAR -> hiển thị ra bằng 1 textarea, staff nhập vô. Vd: 51G-61660, 52F-67558
